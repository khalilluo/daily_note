class Solution:
    def solve(self , a ):
        # write code here
        index = len(a) -1
        while index >= 0:
            if index == len(a) -1 and a[index] > a[index -1]:
                return index
            elif index == 0 and a[index] > a[index + 1]:
                return index
            elif a[index] > a[index -1] and a[index] > a[index + 1]:
                return index
            index -= 1
